\frametitle{Finite Volume Method}
\begin{itemize}
\item The most popular for computational fluid dynamics.
\item Based on Gaussion divergence theorem.
\begin{fleqn}
\begin{equation}
\int_{\Omega} \nabla \cdot \mathbf{u} \ d\Omega = \oint_{S} \mathbf{n} \cdot \mathbf{u} \ dS
\end{equation}
\end{fleqn}


\end{itemize}
