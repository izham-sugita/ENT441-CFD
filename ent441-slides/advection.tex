      \begin{fleqn}
      \begin{equation}
      \frac{\partial u}{\partial t} + c \frac{\partial u}{\partial x} = 0, \qquad
      c \geq 0
      \label{eq:advection}
      \end{equation}
      
      \begin{equation}
      u\left(x,0\right)=u_0 \nonumber
      \label{eq:advection}
      \end{equation}
      \end{fleqn}
\end{itemize}
