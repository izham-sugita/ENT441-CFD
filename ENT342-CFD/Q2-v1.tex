\begin{align}
\nabla \cdot \textbf{u} &= 0 \nonumber  \\ \nonumber
\frac{\partial \textbf{u}}{\partial t} + \textbf{u} \cdot \nabla \textbf{u}  &= -\nabla p + \nu \nabla^2 \textbf{u},
\end{align}
\end{subequations}
